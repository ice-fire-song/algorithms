#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int n,m;
    while(scanf("%d %d",&n,&m)!=EOF)
    {
    	char a[200][200];
    	long long s=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
		}
		          if(a[0][0]==1)   
    		   	  {
    		   	  	  if(a[0][1]==0&&a[1][0]==0&&a[1][1]==0)   s++;
    		   	  }
    		   	  if(a[0][m-1]==1)  
    		   	  {
    		   	  	if(a[0][m-2]==0&&a[1][m-1]==0&&a[1][m-2]==0)   s++;
				  }
				  if(a[n-1][0]==1)  
    		   	  {
    		   	  	if(a[n-1][1]==0&&a[n-2][0]==0&&a[n-2][1]==0)   s++;
				  }
				  if(a[n-1][m-1]==1)  
    		   	  {
    		   	  	if(a[n-1][m-2]==0&&a[n-2][m-2]==0&&a[n-2][m-1]==0)   s++;
				  }
		for(int i=1;i<n-1;i++)
    	{
    		if(a[i][0]==1)  
			{
				 if(a[i-1][1]==0&&a[i][1]==0&&a[i+1][1]==0&&a[i-1][0]==0&&a[i+1][0]==0)  s++;
		    }
		    if(a[i][m-1]==1)
		    {
		    	if(a[i-1][m-2]==0&&a[i][m-2]==0&&a[i+1][m-2]==0&&a[i-1][m-1]==0&&a[i+1][m-1]==0)  s++;
			}
    	}
    	for(int i=1;i<m-1;i++)
    	{
    		if(a[0][i]==1)
    		{
    			if(a[0][i-1]==0&&a[1][i-1]==0&&a[1][i]==0&&a[1][i+1]==0&&a[0][i+1]==0)  s++;
			}
			if(a[n-1][i]==1)
    		{
    			if(a[n-2][i-1]==0&&a[n-2][i]==0&&a[n-2][i+1]==0&&a[n-1][i+1]==0&&a[n-1][i-1]==0)  s++;
			}
		}
		for(int i=1;i<n-1;i++)
		{
			for(int t=1;t<m-1;t++)
			{
				if(a[i][t]==1)
				{
					if(a[i-1][t-1]==0&&a[i-1][t]==0&&a[i-1][t+1]==0&&a[i][t-1]==0&&a[i][t+1]==0&&a[i+1][t-1]==0&&a[i+1][t]==0&&a[i+1][t+1]==0) s++;
				}
			}
		}
		cout<<s<<endl;
	}
}
